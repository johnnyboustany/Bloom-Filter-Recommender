80    public Collection<String> getAttributes() {
81      Collection<String> attributes = new ArrayList<>();
82      for (int i = 2; i <= 13; i++) {
83        attributes.add(studentAttributes[i]);
84      }
85      String comma = ", ";
86  
87      String[] strengthsArr = studentAttributes[14].split(comma);
88      attributes.addAll(Arrays.asList(strengthsArr));
89  
90      String[] weaknessesArr = studentAttributes[15].split(comma);
91      attributes.addAll(Arrays.asList(weaknessesArr));
92  
93      String[] skillsArr = studentAttributes[16].split(comma);
94      attributes.addAll(Arrays.asList(skillsArr));
95  
96      String[] interestsArr = studentAttributes[17].split(comma);
97      attributes.addAll(Arrays.asList(interestsArr));
98  
99      return attributes;
100   }
