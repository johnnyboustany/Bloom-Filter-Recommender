19      // Euclidean distance calculation
20      // Assumes ob1 and ob2 coordinates are same dim
21      double[] ob1Coordinates = ob1.getCartesianCoordinates();
22      double[] ob2Coordinates = ob2.getCartesianCoordinates();
23      double sum = 0;
24      final int NUM_DIMS = ob1Coordinates.length;
25      assert ob2Coordinates.length == NUM_DIMS :
26          " Dimensions must be equal.";
27      for (int i = 0; i < NUM_DIMS; i++) {
28        double a1 = ob1Coordinates[i];
29        double b2 = ob2Coordinates[i];
30        double diff_squared = (a1 - b2) * (a1 - b2);
31        sum += diff_squared;
32      }
33      return Math.sqrt(sum);
34    }
