10  /**
11   * Class to calculate the nearest neighbors (nearest k neighbors).
12   * This class takes in the tree and recursively checks for the nearest neighbors
13   * closest in euclidean distance to a given target.
14   * @param <T>
15   */
16  public class NearestNeighbors<T extends Coordinates> {
17  
18    private kdNode<T> rootNode; // the root node of the tree
19    private final Distance<Coordinates> CUSTOM_USER_DISTANCE = new EuclideanDistance();
20    private int dimensions;
21  
22    public NearestNeighbors(kdNode<T> root_node, T[] data) {
23      this.rootNode = root_node;
24      this.dimensions = data[0].getCartesianCoordinates().length;
25    }
26  
27    /**
28     * Used to calculate the distance between the relative axis.
29     * "If the Euclidean distance between the target point and the farthest of
30     * the current “best neighbors” you have so far is greater than the relevant axis
31     * distance* between the current node and target point, recur on both children." (from the help docs).
32     * This simple method calculates that axis difference.
33     * @param o1
34     * @param o2
35     * @return
36     */
37    public double difference(double o1, double o2) {
38      return Math.abs(o1 - o2);
39    }
40  
41    /**
42     * Need to check BOTH branches:
43     *
44     * "If the previous if-statement is false and you do not need to recur down both children, then:
45     * ● If the current node's coordinate on the relevant axis is less than the target's coordinate
46     * on the relevant axis, recur on the right child.
47     * ● Else if the current node's coordinate on the relevant axis is greater than the target's coordinate on the relevant axis, recur on the left child.
48     * ● (In either case, we are essentially pruning the subtrees that we know with certainty, looking only at the relevant axis distance, will not be any
49     * better than our current nearest neighbors)" - as per the help docs.
50     * This boolean carries out the above three checks to ensure that both left and right
51     * subtrees of the class are iterated through.
52     * @param qs — the size of the priority queue.
53     * @param k — the number of neighbors being searches for.
54     * @param distance — the euclidean distance to the priority node.
55     * @param diff — the difference between the axis.
56     * @return
57     */
58    public boolean checkBranches(int qs, int k, double distance, double diff) {
59      return (distance > diff || qs < k);
60    }

77    /**
78     * Method to recursively get the closest neighbors.
79     * @param priorityNode — the node that is the priority of the search. E.g. if the user input requests
80     *                     the node having id 1, this will be the priority node.
81     * @param k - number of neighbors to find.
82     * @param curr - current node.
83     * @param queue - add the nodes into a priority queue.
84     * @param recursiveDepth - keep track of the depth at which
85     *                       iterations are occurring.
86     */
87    private void nnAlgorithm(Coordinates priorityNode, int k, kdNode<T> curr,
88                             PriorityQueue<Coordinates> queue, int recursiveDepth) {
89      if (curr == null) {
90        return;
91      }
92      if (k <= 0) {
93        return;
94      }
95      // get the coordinates of the current node
96      Coordinates cords = curr.getCords();
97      // add to queue.
98      queue.add(cords);
99      // get the depth of curr
100     double currNodeDepth = curr.getDepth(recursiveDepth);
101     // get depth of target node
102     double pNodeDepth = priorityNode.getCartesianCoordinates()[recursiveDepth];
103     // relative difference
104     double diff = difference(pNodeDepth, currNodeDepth);
105     // distance to target
106     double distance = this.CUSTOM_USER_DISTANCE.distance(priorityNode, queue.peek());
107     // update depth
108     int depth = (recursiveDepth + this.dimensions) % this.dimensions;
109     // BOOLEAN - recur down both subtrees
110     if (checkBranches(queue.size(), k, distance, diff)) {
111       nnAlgorithm(priorityNode, k, curr.getLeft(), queue, depth);
112       nnAlgorithm(priorityNode, k, curr.getRight(), queue, depth);
113     } else {
114       if (currNodeDepth <= pNodeDepth) {
115         nnAlgorithm(priorityNode, k, curr.getRight(), queue, depth);
116       } else {
117         nnAlgorithm(priorityNode, k, curr.getLeft(), queue, depth);
118       }
119     }
120   }

122   /**
123    * Method to add the poll the nodes into a list according to closest to furthest
124    * and to print the results.
125    * @param k
126    * @param userID
127    */
128   public void nnAlgorithmResults(int k, int userID) {
129     try {
130       if (k <= 0) {
131         throw new IllegalArgumentException();
132       }
133       if (this.rootNode == null) {
134         throw new NullPointerException();
135       }
136       Coordinates priorityNode = getPriorityNode(userID, this.rootNode);
137       PriorityQueue<Coordinates> queue = new PriorityQueue<>(Comparator.comparingDouble(
138           curr -> -1 * this.CUSTOM_USER_DISTANCE.distance(priorityNode, curr)));
139       nnAlgorithm(priorityNode, k, this.rootNode, queue, 0);
140       List<Integer> id_list = new ArrayList<>();
141       while (!queue.isEmpty()) {
142         id_list.add(0, Objects.requireNonNull(queue.poll()).id());
143       }
144       int index = 0;
145       id_list.remove(index);
146       for (int i = 0; i < k; i++) {
147         System.out.println(id_list.get(i));
148       }
149     } catch (NullPointerException a) {
150       System.out.println("ERROR:");
151     } catch (NoSuchElementException b) {
152       System.out.println("ERROR:");
153     } catch (IllegalArgumentException c) {
154       System.out.println("ERROR:");
155     }
156   }

158   /**
159    * Helper method to find the node that the user wants
160    * @param userID — inputted user id
161    * @param curr — current node
162    * @return — the coordinates of the node being searched for.
163    */
164   private Coordinates getPriorityNode(int userID, kdNode<T> curr) {
165     // getCord() returns a coordinate object
166     // The coordinate object has method id()
167     // When id is called it should get the id from the Student class which implements coordinate
168     if (curr == null) {
169       // if there is no node left or right.
170       return null;
171     }
172     if (curr.getCords().id() == userID) {
173       return curr.getCords();
174     } else {
175       Coordinates iterate_left = getPriorityNode(userID, curr.getLeft());
176       Coordinates iterate_right = getPriorityNode(userID, curr.getRight());
177       if (iterate_right != null) {
178         return iterate_right;
179       }
180       return iterate_left;
181     }
182   }
