3   import java.math.BigInteger;
4   import java.nio.charset.Charset;
5   import java.nio.charset.StandardCharsets;
6   import java.security.MessageDigest;
7   import java.security.NoSuchAlgorithmException;
8   import java.util.BitSet;
9   
10  /**
11   * BloomFilter contains an array of bits
12   * that it can call methods on.
13   */
14  public class BloomFilter {
15  
16    private static MessageDigest HASH_FUNCTION = null;
17    static {
18      try {
19        HASH_FUNCTION = MessageDigest.getInstance("SHA-1");
20      } catch (NoSuchAlgorithmException e) {
21        e.printStackTrace();
22      }
23    }
24    private static final Charset CHARSET = StandardCharsets.UTF_8;
25    private final int k;
26    private final int m;
27    private final FixedSizeBitSet bits;
28  
29    /**
30     * Constructs the bloom filter.
31     * @param r desired false positive rate
32     * @param n maximum number of elements that will be inserted
33     */
34    public BloomFilter(double r, int n) {
35  
36      // # of hash functions
37      k = (int) Math.ceil(-1 * log2(r));
38  
39      // m bit length
40      m = (int) Math.ceil((k * n) / Math.log(2));
41  
42      bits = new FixedSizeBitSet(m);
43    }
44  
45    /**
46     * Inserts into BF.
47     * @param element string
48     * @return String
49     */
50    public String insertBF(String element) {
51      byte[] b = element.getBytes(StandardCharsets.UTF_8);
52      BigInteger[] hashes = createHashes(b, k);
53      for (int i = 0; i < k; i++) {
54        int key = hashes[i].mod(BigInteger.valueOf(m)).intValue();
55        bits.set(key);
56      }
57      return bits.toString();
58    }
59    /**
60     * Queries BF for the element.
61     * @param element string
62     * @return boolean
63     */
64    public boolean queryBF(String element) {
65      byte[] b = element.getBytes(StandardCharsets.UTF_8);
66      BigInteger[] hashes = createHashes(b, k);
67  
68      for (int i = 0; i < k; i++) {
69        int key = hashes[i].mod(BigInteger.valueOf(m)).intValue();
70  
71        if (!bits.get(key)) {
72          return false;
73        }
74      }
75      return true;
76    }
77    /**
78     * Function to calculate the
79     * log base 2 of an integer.
80     * @param x a double
81     * @return a double (result)
82     */
83    public static double log2(double x) {
84      // calculate log2 x indirectly
85      // using log() method
86      return Math.log(x) / Math.log(2);
87    }
88  
89    /**
90     * Generates hashes based on the contents of an array of bytes, converts the result into
91     * BigIntegers, and stores them in an array. The hash function is called until the required number
92     * of BigIntegers are produced.
93     * For each call to the hash function a salt is prepended to the data. The salt is increased by 1
94     * for each call.
95     *
96     * @param data input data.
97     * @param numHashes number of hashes/BigIntegers to produce.
98     * @return array of BigInteger hashes
99     */
100   public static BigInteger[] createHashes(byte[] data, int numHashes) {
101     BigInteger[] result = new BigInteger[numHashes];
102 
103     int k = 0;
104     BigInteger salt = BigInteger.valueOf(0);
105     while (k < numHashes) {
106       HASH_FUNCTION.update(salt.toByteArray());
107       salt = salt.add(BigInteger.valueOf(1));
108       byte[] hash = HASH_FUNCTION.digest(data);
109       HASH_FUNCTION.reset();
110 
111       // convert hash byte array to hex string, then to BigInteger
112       String hexHash = bytesToHex(hash);
113       result[k] = new BigInteger(hexHash, 16);
114       k++;
115     }
116     return result;
117   }
118 
119   /**
120    * Converts a byte array to a hex string.
121    * Source: https://stackoverflow.com/a/9855338
122    *
123    * @param bytes the byte array to convert
124    * @return the hex string
125    */
126   private static String bytesToHex(byte[] bytes) {
127     byte[] hexArray = "0123456789ABCDEF".getBytes(CHARSET);
128     byte[] hexChars = new byte[bytes.length * 2];
129     for (int j = 0; j < bytes.length; j++) {
130       int v = bytes[j] & 0xFF;
131       hexChars[j * 2] = hexArray[v >>> 4];
132       hexChars[j * 2 + 1] = hexArray[v & 0x0F];
133     }
134     return new String(hexChars, CHARSET);
135   }

137 
138   /** Getter to return the bitset.
139    *
140    * @return bitSet
141    */
142   public BitSet getBitSet() {
143     return bits;
144   }
