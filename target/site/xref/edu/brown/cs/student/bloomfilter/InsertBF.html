17    /** Checks if the insert_bf command has been called correctly.
18     *
19     * @param commandInputs passed in from the BFCommands class
20     */
21    @Override
22    public void run(String[] commandInputs) {
23      if (bf != null && commandInputs.length == 2) {
24        System.out.println(bf.insertBF(commandInputs[1]));
25      } else {
26        System.out.println("ERROR: Invalid insert_bf Command");
27      }
28    }

36    }
