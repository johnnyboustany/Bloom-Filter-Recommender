10  /** Recommends fishes based on the given fish
11   * using a similarity algorithm.
12   */
13  public class BFRecommender {
14  
15    private final List<Student> fishes;
16    private final SimilarityMeasure similarity;
17  
18    /** Stores the fishes data structure and similarity measure
19     * for use in the similar fishes algorithm.
20     * @param studs fishes
21     * @param sim similarity algorithm
22     */
23    public BFRecommender(List<Student> studs, SimilarityMeasure sim) {
24      fishes = studs;
25      similarity = sim;
26    }
27  
28    /** Finds the fish based on their userID.
29     *
30     * @param userId of fish
31     * @return Student that the id belongs to
32     */
33    public Student findStudent(int userId) {
34  
35      for (Student fish : fishes) {
36        if (fish.getId() == userId) {
37          return fish;
38        }
39      }
40      return null;
41    }
42    /** Runs the similar fishes algorithm to find a k number of most similar
43     * fishes to a specific given fish.
44     *
45     * @param k integer number of similar fishes
46     * @param user_id of given fish
47     * @return Collection<Student> an arraylist containing the k most similar
48     * fishes to the given fish
49     */
50    public Collection<Student> findSimilarStudents(int k, int user_id) {
51      //System.out.println(fishes);
52      if (findStudent(user_id) != null) {
53        PriorityQueue<Student> pq = new PriorityQueue<>(new BFComparator());
54  
55        // if a name was given, then that fish has to be skipped and not
56        // considered as a similar fish
57        boolean skippedAStudent = false;
58        for (Student fish : fishes) {
59          if (!skippedAStudent && fish.getId() == user_id) {
60            skippedAStudent = true;
61            // System.out.println(fish.getId());
62            continue;
63          }
64          fish.setNumTrueBits(findStudent(user_id), similarity);
65          pq.add(fish);
66        }
67  
68        Collection<Student> similarFishes = new ArrayList<>();
69  
70        // if k is larger than the number of stars being stored,
71        // then all of them are added to the similarFishes arrayList,
72        // except the fish that a userID was given for
73  
74        for (int i = 0; i < k; i++) {
75          // removing the top of the queue
76          // and storing it in s
77          Student s = pq.poll();
78          if (s != null) {
79            similarFishes.add(s);
80          }
81        }
82        return similarFishes;
83  
84      } else {
85        return null;
86      }
87    }
88  }
