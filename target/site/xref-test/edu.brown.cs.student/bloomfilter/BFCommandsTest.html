13    @Test
14    public void simpleBFCommandsTest() {
15  
16      BFCommands commands = new BFCommands();
17      assertTrue(commands.getBloomFilter() == null);
18      String command1 = "create_bf 0.1 3";
19      String[] lineArr1 = command1.split(" ");
20      commands.run(lineArr1);
21  
22      assertTrue(commands.getBloomFilter() != null);
23  
24      BloomFilter bf = commands.getBloomFilter();
25      assertEquals(bf.getBitSet().toString(), "000000000000000000");
26  
27      String command2 = "insert_bf dog";
28      String[] lineArr2 = command2.split(" ");
29      commands.run(lineArr2);
30      assertEquals(bf.getBitSet().toString(), "000000110000110000");
31  
32      String command3 = "query_bf dog";
33      String[] lineArr3 = command3.split(" ");
34      commands.run(lineArr3);
35      assertTrue(bf.queryBF("dog"));
36    }

39    @Test
40    public void CommandClassesTest() {
41  
42      CreateBF createBF = new CreateBF();
43      String command1 = "create_bf 0.1 3";
44      String[] lineArr1 = command1.split(" ");
45      createBF.run(lineArr1);
46      BloomFilter bf = createBF.getBloomFilter();
47      assertNotNull(bf);
48  
49      InsertBF insertBF = new InsertBF();
50      String command2 = "insert_bf dog";
51      String[] lineArr2 = command2.split(" ");
52      // didn't set the bf
53      insertBF.run(lineArr2);
54      assertEquals(bf.getBitSet().toString(), "000000000000000000");
55  
56      // now setting the bf
57      insertBF.setBloomFilter(bf);
58      insertBF.run(lineArr2);
59      assertEquals(bf.getBitSet().toString(), "000000110000110000");
60  
61      QueryBF queryBF = new QueryBF();
62      String command3 = "query_bf dog";
63      String[] lineArr3 = command3.split(" ");
64  
65      // didn't set the bf
66      queryBF.run(lineArr3);
67      assertNull(queryBF.getBloomFilter());
68  
69      // now setting the bf
70      queryBF.setBloomFilter(bf);
71      queryBF.run(lineArr3);
72      assertTrue(bf.queryBF("dog"));
73    }
